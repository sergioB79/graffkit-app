function buildPrompt(config) {
  return `
Graffiti street art with the word "${config.text}",
${config.style} lettering,
complexity level ${config.complexity}/10,
${config.lineStyle} lines,
color palette: ${config.colors},
features: ${config.features.join(", ")},
strong outline, urban spray paint texture,
${config.background} background,
high detail, professional graffiti, centered composition
  `;
}

🧪 EXEMPLO API (OpenAI Images)
// /api/generate.js
import OpenAI from "openai";

const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY
});

export default async function handler(req, res) {
  const { config } = req.body;

  const prompt = buildPrompt(config);

  try {
    const result = await client.images.generate({
      model: "gpt-image-1",
      prompt,
      size: "1024x1024"
    });

    res.status(200).json({
      image: result.data[0].url
    });

  } catch (err) {
    res.status(500).json({ error: err.message });
  }
}

async function generateGraffiti(config) {
  const res = await fetch("/api/generate", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ config })
  });

  const data = await res.json();
  return data.image;
}
